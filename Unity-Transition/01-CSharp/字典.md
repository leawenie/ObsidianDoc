## 创建和初始化
```C#
//创建
Dictionary<string, int> items = new Dictionary<string, int>();

//初始化
Dictionary<string, int> backpackItems = new Dictionary<string, int>{
	{"宝剑", 1}, {"长刀", 1}
```

## 添加和访问元素
```C#
Dictionary<string, int> items = new Dictionary<string, int>();
items.Add("宝剑", 1);
items["宝剑"] = 2; //存在该键则覆盖，不存在则赋值

int count = items["大刀"]; //抛出异常

//安全访问
if(items.TryGetValue("大刀", out int count2)){
	Console.WriteLine($"大刀的数量为 {count2}");
}else{
	Console.WriteLine("键不存在");
}

//存在检查
//检查键是否存在
if (items.ContainsKey("宝剑"))
{
    items["宝剑"] =2;  // 更新
}
//检查值是否存在
if (items.ContainsValue(2))
{
    Console.WriteLine("存在");
}
```

## 遍历
```C#
var dict = new Dictionary<string, string>
{
    {"Name", "张三"},
    {"City", "北京"},
    {"Job", "程序员"}
};

// 遍历所有键值对
foreach (KeyValuePair<string, string> kvp in dict)
{
    Console.WriteLine($"键: {kvp.Key}, 值: {kvp.Value}");
}

// 只遍历键
foreach (string key in dict.Keys)
{
    Console.WriteLine($"键: {key}");
}

// 只遍历值
foreach (string value in dict.Values)
{
    Console.WriteLine($"值: {value}");
}
```

## 删除元素
```C# 
var dict = new Dictionary<int, string>
{
    {1, "One"},
    {2, "Two"},
    {3, "Three"}
};

// 删除指定键
bool removed = dict.Remove(2);  // 返回 true/false

// 删除并获取值（C# 7.0+）
if (dict.Remove(3, out string removedValue))
{
    Console.WriteLine($"已删除: {removedValue}");
}

// 清空所有元素
dict.Clear();
```
